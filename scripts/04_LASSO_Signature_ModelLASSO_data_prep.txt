# Prepare input matrix for LASSO regression
library(tidyverse)

# Load normalized expression matrix
expr_data <- read.csv("GSE25066_symbol_avg_filtered_ordered.csv", row.names = 1, check.names = FALSE)
transposed_expr <- as.data.frame(t(expr_data)) %>% tibble::rownames_to_column("sample")

# Load candidate gene list from Venn intersection
venn_genes <- readLines("venn_gene.txt")
existing_genes <- intersect(venn_genes, colnames(transposed_expr))
venn_genes_filtered <- venn_genes[venn_genes %in% existing_genes]

# Subset candidate gene expression
selected_expr <- transposed %>% dplyr::select(sample, all_of(venn_genes_filtered))

# Merge clinical grouping label
group_info <- read.csv("GSE25066_group2.csv", col.names = c("sample", "outcome"))
merged_data <- selected_expr %>% left_join(group_info, by = "sample") %>% dplyr::select(sample, outcome, everything())

# Set sample name as row name
final_data <- merged_data %>% tibble::column_to_rownames("sample")

# Export LASSO input table
write.csv(final_data, "GSE25066-LASSO1.csv", row.names = TRUE, na = "")
