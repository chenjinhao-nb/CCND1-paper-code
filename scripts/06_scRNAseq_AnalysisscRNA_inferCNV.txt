library(rjags)
library(infercnv)
library(Seurat)
library(tidyverse)
library(dplyr)
library(ggplot2)

# Load pre-treatment epithelial Seurat object
seu_epi_pre <- readRDS("seu_epi_pre.rds")

# Set output directory
output_dir <- "InferCNV_Results"
infercnv_dir <- file.path(output_dir, "InferCNV_Analysis")
dir.create(infercnv_dir, recursive = TRUE, showWarnings = FALSE)

# Load input files for inferCNV
infercnv_gene_order <- read.table("infercnv_gene_order.txt", header = FALSE, sep = "\t")
infercnv_count_matrix <- read.table("infercnv_count_matrix.txt", header = TRUE, sep = "\t", row.names = 1)
infercnv_cell_annotations <- read.table("infercnv_cell_annotations.txt", header = FALSE, sep = "\t")
colnames(infercnv_cell_annotations) <- c("cell_id", "cell_type")

# Temp annotation file for inferCNV format
temp_annot_file <- file.path(infercnv_dir, "temp_cell_annot.txt")
write.table(infercnv_cell_annotations, temp_annot_file, sep = "\t", quote = FALSE, row.names = FALSE, col.names = FALSE)

# Create inferCNV object (reference: T cells as normal diploid)
infercnv_obj <- CreateInferCNVObject(
  raw_counts_matrix = "infercnv_count_matrix.txt",
  annotations_file = temp_annot_file,
  delim = "\t",
  gene_order_file = "infercnv_gene_order.txt",
  ref_group_names = c("T_cells")
)

# Run inferCNV pipeline
infercnv_obj <- infercnv::run(
  infercnv_obj,
  cutoff = 0.1,
  out_dir = infercnv_dir,
  cluster_by_groups = TRUE,
  denoise = TRUE,
  HMM = FALSE,
  num_threads = 6,
  analysis_mode = "subclusters"
)
# Remove temporary annotation file
if(file.exists(temp_annot_file)) file.remove(temp_annot_file)

# Save full inferCNV object
saveRDS(infercnv_obj, file = file.path(infercnv_dir, "infercnv_final_obj.rds"))

# Load CNV signal matrix and calculate cell-level CNV score
obs_file <- file.path(infercnv_dir, "infercnv.observations.txt")
if(file.exists(obs_file)) {
  cnv_scores <- read.table(obs_file, header = TRUE, row.names = 1)
  cell_cnv_means <- colMeans(abs(cnv_scores))
  # Set threshold based on epithelial cell CNV score upper quartile
  epithelial_cells <- infercnv_cell_annotations$cell_id[infercnv_cell_annotations$cell_type == "Epithelial"]
  epithelial_cnv <- cell_cnv_means[names(cell_cnv_means) %in% epithelial_cells]
  cnv_threshold <- quantile(epithelial_cnv, 0.75)
  # Classify malignant epithelial cells
  malignant_cells <- names(epithelial_cnv[epithelial_cnv > cnv_threshold])
  seu_epi_pre$Cell_Status <- ifelse(colnames(seu_epi_pre) %in% malignant_cells, "Malignant", "Non_Malignant")
  
  # Violin plot for CNV score distribution
  cnv_plot_data <- data.frame(
    cell_id = names(cell_cnv_means),
    cnv_score = cell_cnv_means,
    cell_type = infercnv_cell_annotations$cell_type[match(names(cell_cnv_means), infercnv_cell_annotations$cell_id)]
  )
  p_cnv <- ggplot(cnv_plot_data, aes(x = cell_type, y = cnv_score, fill = cell_type)) +
    geom_violin(alpha = 0.7) +
    geom_boxplot(width = 0.2, alpha = 0.7) +
    geom_hline(yintercept = cnv_threshold, linetype = "dashed", color = "red") +
    labs(title = "CNV Score Distribution by Cell Type", 
         subtitle = paste("Cutoff threshold:", round(cnv_threshold, 3)),
         x = "Cell Type", y = "Mean Absolute CNV Score") +
    theme_minimal()
  ggsave(file.path(infercnv_dir, "CNV_score_distribution.pdf"), p_cnv, width = 8, height = 6)
  
  # Export CNV score table
  cnv_results <- data.frame(
    cell_id = names(cell_cnv_means),
    cnv_score = cell_cnv_means,
    cell_type = cnv_plot_data$cell_type,
    is_malignant = names(cell_cnv_means) %in% malignant_cells
  )
  write.csv(cnv_results, file.path(infercnv_dir, "cell_cnv_scores.csv"))
  
  cat("Malignant epithelial cell count:", sum(seu_epi_pre$Cell_Status == "Malignant"), "\n")
  cat("Normal epithelial cell count:", sum(seu_epi_pre$Cell_Status == "Non_Malignant"), "\n")
}

# Save epithelial object with malignant cell label
saveRDS(seu_epi_pre, file.path(output_dir, "seu_epi_pre_with_CNV.rds"))
