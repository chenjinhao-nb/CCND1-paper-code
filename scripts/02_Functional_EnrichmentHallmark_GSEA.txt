# Clear environment
rm(list=ls())

# Load packages
library(clusterProfiler)
library(ggplot2)
library(enrichplot)
library(DOSE)
library(org.Hs.eg.db)
library(gridExtra)

# Create output folder
if (!dir.exists("hall")) dir.create("hall")

# Load differential expression table
diff_data <- read.delim("GSE25066_diff.txt", header = TRUE, stringsAsFactors = FALSE)

# Remove duplicated gene entries (retain max absolute logFC)
if(any(duplicated(diff_data$gene_symbol))) {
  diff_data <- diff_data[order(diff_data$gene_symbol, -abs(diff_data$logFC)), ]
  diff_data <- diff_data[!duplicated(diff_data$gene_symbol), ]
}

# Sort gene list descending by log2FC for GSEA input
gene_list <- diff_data$logFC
names(gene_list) <- diff_data$gene_symbol
gene_list <- sort(gene_list, decreasing = TRUE)

# Load MSigDB Hallmark gene set
hallmark_gene_sets <- read.gmt("h.all.v2024.1.Hs.symbols.gmt")
set.seed(123)

# Run GSEA
gsea_res <- GSEA(
  geneList     = gene_list,
  TERM2GENE    = hallmark_gene_sets,
  minGSSize    = 15,
  maxGSSize    = 500,
  pvalueCutoff = 0.05,
  pAdjustMethod = "BH",
  verbose      = FALSE
)

# Export full GSEA result
write.csv(gsea_res@result, "hall/GSEA_Hallmark_full_results.csv", row.names = FALSE)
# Subset significant pathways (adj.P<0.05)
sig_res <- subset(gsea_res@result, p.adjust < 0.05)

if(nrow(sig_res) > 0) {
  # Prepare top10 pathways dotplot data
  dotplot_data <- head(sig_res[order(-abs(sig_res)), ], 10)
  dotplot_data$Description <- gsub("HALLMARK_", "", dotplot_data$Description)
  
  p_dot <- ggplot(dotplot_data, 
                  aes(x = NES, y = reorder(Description, NES), 
                      color = -log10(p.adjust), size = setSize)) +
    geom_point() +
    scale_color_gradient(low = "blue", high = "red", name = "-log10(adjusted P)") +
    labs(x = "Normalized Enrichment Score (NES)", 
         y = NULL,
         title = "Top 10 Significantly Enriched Hallmark Pathways",
         size = "Gene Set Size") +
    theme_bw(base_size = 12) +
    theme(axis.text.y = element_text(size = 10))
  
  pdf("hall/GSEA_top10_dotplot.pdf", width = 10, height = 6)
  print(p_dot)
  dev.off()
  tiff("hall/GSEA_top10_dotplot.tiff", width = 2000, height = 1200, res = 300)
  print(p_dot)
  dev.off()
  
  # Single top pathway enrichment curve
  top_pathway <- sig_res$ID[1]
  pathway_title <- gsub("HALLMARK_", "", sig_res$Description[1])
  p_enrich <- enrichplot::gseaplot2(
    gsea_res, 
    geneSetID = top_pathway,
    title = pathway_title,
    pvalue_table = TRUE,
    color = "firebrick",
    base_size = 12
  )
  pdf("hall/GSEA_top1_single_path.pdf", width = 14, height = 6)
  print(p_enrich)
  dev.off()
  tiff("hall/GSEA_top1_single_path.tiff", width = 3500, height = 1500, res = 300)
  print(p_enrich)
  dev.off()
  
  # Combine top3 pathways curve
  top3 <- sig_res$ID[1:min(3, nrow(sig_res))]
  p_multi <- enrichplot::gseaplot2(
    gsea_res, 
    geneSetID = top3,
    pvalue_table = TRUE,
    base_size = 10,
    color = c("#E41A1C", "#377EB8", "#4DAF4A")
  )
  pdf("hall/GSEA_top3_combined.pdf", width = 14, height = 8)
  print(p_multi)
  dev.off()
  tiff("hall/GSEA_top3_combined.tiff", width = 3500, height = 2000, res = 300)
  print(p_multi)
  dev.off()
  
  # Gene-concept network
  p_net <- cnetplot(gsea_res, 
                    categorySize = "pvalue",
                    foldChange = gene_list,
                    showCategory = 5,
                    colorEdge = TRUE,
                    node_label = "all") +
    ggtitle("Gene-Pathway Interaction Network (Hallmark)") +
    theme(legend.position = "right")
  pdf("hall/GSEA_gene_network.pdf", width = 12, height = 10)
  print(p_net)
  dev.off()
  tiff("hall/GSEA_gene_network.tiff", width = 3000, height = 2500, res = 300)
  print(p_net)
  dev.off()
  
} else {
  message("No significantly enriched Hallmark pathways (adjusted P < 0.05)")
}
