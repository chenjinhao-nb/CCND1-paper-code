# Clear workspace
rm(list=ls())

# Load libraries
library(pheatmap)
library(tidyverse)
library(RColorBrewer)

# Load normalized expression & DEG result
load("GSE25066.rda")
DEG <- read.table("GSE25066_diff1.txt", sep = "\t", header = TRUE, row.names = 1,check.names = FALSE)
stopifnot(identical(rownames(group), colnames(expr)))

# Filter significant DEGs
sig_genes <- rownames(DEG)[DEG$group != "NOT"]
exp_diff <- expr[sig_genes, ]

# Select top25 up-regulated & top25 down-regulated genes
DEG_filtered <- DEG[DEG$group != "NOT", ]
top_up <- DEG_filtered %>% arrange(desc(log2FC)) %>% head(25)
top_down <- DEG_filtered %>% arrange(log2FC) %>% head(25)
selected_genes <- c(rownames(top_up), rownames(top_down))
exp_diff2 <- expr[selected_genes, ]

# Group color annotation
annotation_colors <- list(response = c(NonpCR="#E64B35FF",pCR="#4DBBD5FF"))

# Draw publication heatmap
final_heatmap <- pheatmap(
  exp_diff2,
  annotation_col = group,
  annotation_colors = annotation_colors,
  color = colorRampPalette(rev(brewer.pal(11, "RdBu")))(100),
  scale = "row",
  cluster_cols = FALSE,
  cluster_rows = FALSE,
  show_colnames = FALSE,
  show_rownames = TRUE,
  border_color = NA,
  fontsize_row = 7,
  cellwidth = 2,
  cellheight = 10,
  gaps_row = 25,
  main = "Top 50 Differentially Expressed Genes",
  legend = TRUE
)

# Save figure
sample_count <- ncol(exp_diff2)
ggsave("DEG_top50_heatmap.pdf",
       plot = final_heatmap,
       width = max(6, sample_count * 0.15),
       height = 20,
       dpi = 300,
       units = "cm")

ggsave("DEG_top50_heatmap.tiff",
       plot = final_heatmap$gtable,
       width = max(6, sample_count * 0.15),
       height = 20,
       dpi = 300,
       units = "cm",
       device = "tiff")
