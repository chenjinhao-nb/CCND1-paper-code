library(Seurat)
library(tidyverse)

# Load pre-treatment epithelial object with CNV malignant label
seu_epi_pre <- readRDS("seu_epi_pre_with_CNV.rds")
# Subset only malignant epithelial cells
seu_epi_malignant <- subset(seu_epi_pre, subset = Cell_Status == "Malignant")
cat("Total malignant epithelial cells:", ncol(seu_epi_malignant), "\n")
saveRDS(seu_epi_malignant, "seu_epi_malignant.rds")

# UMAP plot for CCND1 expression
ccnd1_umap <- FeaturePlot(
  object = seu_epi_malignant,
  features = "CCND1",
  reduction = "umap",
  pt.size = 0.5,
  order = TRUE
) +
  ggtitle("CCND1 Expression in Malignant Epithelial Cells") +
  theme(plot.title = element_text(hjust = 0.5, face = "bold")) +
  scale_color_gradient(low = "#4DBBD5FF", high = "#E64B35FF")
ggsave("CCND1_malignant_umap.pdf", ccnd1_umap, width = 8, height = 8)

# Extract scaled expression matrix for correlation
expr_matrix <- GetAssayData(seu_epi_malignant, assay = "RNA", layer = "scale.data")

gene_expr_pct <- apply(expr_matrix, 1, function(x) sum(x != 0) / length(x))
gene_expr_var <- apply(expr_matrix, 1, function(x) var(x, na.rm = TRUE))
